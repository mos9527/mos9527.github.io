# JC0301. Angry Cows

**分析:** 单调性在对于牛舍距离，大于等于最小值$x$的距离可取，反之不行

上板子即可

**CODE:** 

```c++
#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <algorithm>
#include <numeric>
#include <vector>
#include <stdio.h>
#include <cmath>
#include <queue>
using namespace std;
typedef long long ll;
typedef vector<ll> v;
#define PRED(X) [](auto const& lhs, auto const& rhs) {return X;}
#define PREDT(T,X) [](T const& lhs, T const& rhs) {return X;}
#define SUM(T,X)[](ll sum, T const& elem) { return X;}
#define PAIR2(T) pair<T,T>
ll A[(ll)1e6];
int main() {
    // std::ios::sync_with_stdio(false);
    ll n, m; cin >> n >> m;
    for (ll i = 1; i <= n; i++) {
        scanf("%lld", &A[i]);
    }
    sort(A + 1, A + n + 1);
    auto count = [&](ll d) {
        ll cnt = 1, prev = A[1];
        for (ll i = 1; i <= n; i++) {
            if (A[i] - prev >= d)
                prev = A[i], cnt++;
        }
        return cnt;
    };
    ll l = A[1], r = A[n], ans = 0;
    while (l < r)
    {
        ll mid = (l + r + 1) / 2;
        if (count(mid) >= m) l = mid, ans = max(ans,mid);
        else r = mid - 1;
    }
    cout << ans;
    return 0;
}
```

### JC0302. Best Cow Fences

**分析：** 直接观察数据并不能找到容易利用的单调性；但考虑平均数需要求和..

嗯，可以考虑**构造前缀和**嘛

思路在对均值$avg$进行check，看看能不能找到满足长度和$avg$的子段

check时构造前缀和，这时可以直接$-avg$；如此就有了**如果存在这样的子段，那么这样的子段和>0**的性质

**CODE:**

```c++

#include <algorithm>
#include <numeric>
#include <vector>
#include <stdio.h>
#include <cmath>
#include <queue>
using namespace std;

typedef vector<ll> v;
#define EPS 0.0001F
#define DIMENSION 1e6
#define DIM (size_t)(DIMENSION)
#define PRED(X) [](auto const& lhs, auto const& rhs) {return X;}
#define PREDT(T,X) [](T const& lhs, T const& rhs) {return X;}
#define SUM(T,X)[](ll sum, T const& elem) { return X;}
#define PAIR2(T) pair<T,T>
double A[DIM], temp[DIM];
int main() {
    // std::ios::sync_with_stdio(false);
    ll n, len; cin >> n >> len;
    for (ll i = 1; i <= n; i++)
        scanf("%lf", &A[i]);    
    auto check = [&](double avg) {
        for (ll i = 1; i <= n; i++)
            temp[i] = temp[i - 1] + A[i] - avg;
        double mmin = 0;
        for (ll l = len; l <= n; l++) {
            mmin = min(mmin, temp[l - len]);
            if (temp[l] - mmin >= 0)
                return true;
        }
        return false;
    };
    double l = 0, r = 2001, ans = 0;
    while (r - l > EPS) {
        double mid = (l + r) / 2;
        if (check(mid)) {
            l = mid;
            ans = max(ans, mid);
        }
        else
            r = mid;
    }
    printf("%d", (int)(r * 1000));
    return 0;
}
```

### JC0303. Chain Reaction

**思路：**这题看[教程](https://codeforces.com/blog/entry/22256)了...

递推式如下，记$B[i]$为$i$处信标的破坏力：

$dp[i] = \begin{cases}dp[i-1] & \text{i 处没有信标}\\dp[i - B[i] - 1] + 1 & \text{i 处有信标}\end{cases}$

好奇这题上二分能怎么解...

**CODE：**

```c++
#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <algorithm>
#include <numeric>
#include <vector>
#include <stdio.h>
#include <cmath>
#include <queue>
using namespace std;
typedef long long ll;
typedef double lf;
typedef vector<ll> v;
#define EPS 0.0001F
#define PRED(X) [](auto const& lhs, auto const& rhs) {return X;}
#define PREDT(T,X) [](T const& lhs, T const& rhs) {return X;}
#define SUM(T,X)[](ll sum, T const& elem) { return X;}
#define PAIR2(T) pair<T,T>

#define DIMENSION 1e7
#define DIM (size_t)(DIMENSION)
ll M[DIM], dp[DIM] /* not destroyed! */;
int main() {
    // std::ios::sync_with_stdio(false);
    ll n; cin >> n;
    ll dim = 0;
    for (ll i = 1, a, b; i <= n; i++) {
        cin >> a >> b;
        M[a] = b;
        dim = max(dim, a);
    }
    ll m = 0, ians = 0;
    if (M[0]) dp[0] = 1;
    for (ll i = 1; i < DIM; i++) {
        if (M[i] == 0) {
            dp[i] = dp[i - 1];
        }
        else if (M[i] >= i) {
            dp[i] = 1;
        }
        else {
            dp[i] = dp[i - M[i] - 1] + 1;
        }
        ians = max(ians, dp[i]);
    }
    cout << n - ians;
    return 0;
}
```

### JC0304. Chat Ban

**分析：** 嗯..等差数列

碰运气写出来个$O(1)$的答案（逃

分段推解等差数列求和项数可以直接找到被ban的上界；而解这样的二次方程则是很简单的事情了

```c++
int main() {
    // std::ios::sync_with_stdio(false);
    ll t; cin >> t;
    const auto solve1 = [](ll x) {
        return ceill((-1 + sqrtl(1 + 8 * x)) / 2.0);
    };
    const auto solve2 = [](ll k, ll xr) {
        ll t = floorl((1 + sqrtl(1 + 4 * (k * k - k - 2 * xr))) / 2.0);
        return k + k - 1 - t + 1;
    };
    while (t--) {
        ll k, x; cin >> k >> x;
        ll half = (k * (k - 1)) / 2;
        ll mid = k;
        ll ans = 0;
        if (x <= half + mid) {
            ans = solve1(x);
            while (ans * (ans + 1) / 2 < x) ans++;
        }
        else 
            if (half * 2 + mid <= x) {
                ans = 2 * k - 1;                
            }
            else {
                ans = solve2(k, x - half - mid);    
            }
        cout << ans << '\n';
    }
    return 0;
}
```

### JC0308. Keshi Is Throwing a Party

**分析：** 直接看query要怎么写吧！这里是个贪心

记朋友$i$乐意有不多于$A_i$个比他穷，$B_i$个比他富的人在场

显然，若一共邀请$x$人，对朋友$i$，最多会有$A_i$个比他穷，$x - A_i - 1$个比他富的人也在场

找到最穷的,满足上述条件的朋友，即可满足局部最优条件

然后套二分板子就ok

**CODE:**

```c++
#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <algorithm>
#include <numeric>
#include <vector>
#include <stdio.h>
#include <cmath>
#include <queue>
using namespace std;
typedef long long ll;
typedef double lf;
typedef vector<ll> v;
#define EPS 0.0001F
#define PRED(X) [](auto const& lhs, auto const& rhs) {return X;}
#define PREDT(T,X) [](T const& lhs, T const& rhs) {return X;}
#define SUM(T,X)[](ll sum, T const& elem) { return X;}
#define PAIR2(T) pair<T,T>

#define DIMENSION 1e6
#define DIM (size_t)(DIMENSION)
ll n, A[DIM], B[DIM];
bool query(ll num) {
    ll invite = 0;
    for (ll i = 1; i <= n; i++) {
        ll poor = A[i];
        ll rich = num - poor - 1;
        if (rich <= invite && invite <= B[i]) invite++;
    }
    return invite >= num;
}
int main() {
    // std::ios::sync_with_stdio(false);
    ll t; cin >> t;
    while (t--) {
        cin >> n;
        for (ll i = 1; i <= n; i++) scanf("%lld%lld", &A[i], &B[i]);
        ll l = 1, r = n + 1, ans = 0;
        while (l < r)
        {
            ll mid = (l + r + 1) / 2;
            if (query(mid)) l = mid;
            else r = mid - 1;
        }
        cout << l << '\n';
    }
    return 0;
}
```

### JC0312. Set or Decrease

**分析：** 只有两个操作：让某个数$a_i$减小$1$或让某两个数$a_i,a_j$相等；直到数列和$\leq k$

显然，把最小的数减小后然后复制给别的数是最划算的；但显然操作数不能只花在这里

那么，设这个操作进行$x$次，复制进行$y$次；问题即求$min(x,y)$

毕竟要求和，做个前缀和即为$P_i$吧！一共$n$个数

操作后，数列和即为$sum = (1+y)(a_1 - x) + P_{n-y} - A_1$

令$sum$最接近$k$最划算，解$sum \leq k$

得$A1 - x \leq (k+A_1-P_{n-y})/(1+y)$

即$x = A1 - \lfloor((k+A_1-P_{n-y})/(1+y))\rfloor$

之后搜索即可

原来不用二分也够快（

**CODE:**

```C++
#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <algorithm>
#include <numeric>
#include <vector>
#include <stdio.h>
#include <cmath>
#include <queue>
using namespace std;
typedef long long ll;
typedef double lf;
typedef vector<ll> v;
template<typename T> const T sgn(T const& v) {
    if (v > 0) return 1;
    if (v < 0) return -1;
    return 0;
}
#define EPS 0.0001F
#define PRED(X) [](auto const& lhs, auto const& rhs) {return X;}
#define PREDT(T,X) [](T const& lhs, T const& rhs) {return X;}
#define SUM(T,X)[](ll sum, T const& elem) { return X;}
#define PAIR2(T) pair<T,T>

#define DIMENSION 1e6
#define DIM (size_t)(DIMENSION)
ll A[DIM];
ll P[DIM];
int main() {
    // std::ios::sync_with_stdio(false);
    ll t; cin >> t;
    while (t--) {
        ll n, k; cin >> n >> k;
        ll sum = 0;
        for (ll i = 1; i <= n; i++) scanf("%lld", A + i);        
        sort(A + 1, A + n + 1);
        for (ll i = 1; i <= n; i++) P[i] = A[i] + P[i - 1];
        // decrease A1 x times
        // set biggest y numbers to A1
        // ans = min(x+y)
        // with op(x,y) we have
        // sum = (1+y)(A1 - x) + P[n-y] - P[1]
        // sum <= k. solve for x:
        // A1 - x \leq (k+P[1]-P[n-y])/(1+y)
        // min x -> x = A1 - \floor((k+P[1]-P[n-y])/(1+y))
        ll ans = 1e10;
        for (ll y = 0; y < n; y++) {
            ll t = k + P[1] - P[n - y];
            ll x = A[1] - floor(t / (double)(1 + y));
            ans = min(ans, max(0LL, max(0LL,x) + y));
        }
        cout << ans << '\n';
    }
    return 0;
}
```

### JC0313. Valhalla Siege

**分析：** `std::upper_bound`板子题

把箭数求前缀和后，问题即转化成$i$时这样的伤害总和会击倒多少战士

额外地，还需要考虑全军覆没的情况

```
#include <vector>

#define PREDT(T,X) [](T const& lhs, T const& rhs) {return X;}

#define PAIR2(T) pair<T,T>

#define DIM (size_t)(DIMENSION)
ll A[DIM];
ll P[DIM];
ll K[DIM];
int main() {
    // std::ios::sync_with_stdio(false);
    ll n, q; cin >> n >> q;
    for (ll i = 1; i <= n; i++) {
        scanf("%lld", A + i);
        P[i] = P[i - 1] + A[i];       
    }
    for (ll i = 1; i <= q; i++)
        scanf("%lld", K + i);
    ll qsum = 0;
    for (ll t = 1; t <= q; t++) {       
        qsum += K[t];
        ll fell = upper_bound(P + 1, P + 1 + n, qsum, std::less<ll>()) - P;
        if (fell - 1 >= n) {
            qsum = 0;
            fell = 1;
        }
        cout << n - fell + 1 << '\n';
    }
    return 0;
}
```
