## #JC0102. Dreamoon and Ranking Collection

---

**分析：** 给定正整数数列$a_n$和和$x$，填充$x$个数让$a_n$从1开始连续的上限值

去重排序后插入即可；使用`set`可以一步搞定

**Code:** 

```c++
#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <algorithm>
#include <numeric>
#include <vector>
#include <stdio.h>
#include <cmath>
#include <set>
using namespace std;
typedef long long ll;
typedef vector<ll> v;
int main() {
    // std::ios::sync_with_stdio(false);
    ll t;
    cin >> t;
    while (t--) {
        ll n, x; cin >> n >> x;
        set<ll> A; for (ll i = 0; i < n; i++) {
            ll a; cin >> a;
            A.insert(a);
        }
        ll ans = -1;
        for (ll i = 1;;i++) {
            if (A.find(i) == A.end()) {
                if (x == 0) break;
                x--;
            }
            ans = max(ans, i);
        }
        cout << ans << '\n';
    }
    return 0;
}
```

### #JC0104. Happy Birthday, Polycarp!

---

**分析：** 找到满足正整数$m\in[1,n]$, $m$的每个数位都一致的$m$的数量

- 数学归纳法可知当 $n\leq9$,数量为$n$

- 当$n\geq10$,以 $m\in[10,100]$ 可以发现有 ${11,22,33,44,55,66,77,88,99}$ 符合条件

​	可以推广到 $m\in[10^p,10^{p+1}]$ ，之中有 $9p$ 个数符合条件

- 当$m\in[10^p,10^{p+1})$ ,*可以*通过$m$内最小数位直接得到这个区间的数个数

  *或者除以长$p+1$的 $111...$向下取整（*

**Code:**

```c++
#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <algorithm>
#include <numeric>
#include <vector>
#include <stdio.h>
#include <cmath>
using namespace std;
typedef long long ll;
typedef vector<ll> v;
int main() {
    // std::ios::sync_with_stdio(false);
    ll t; cin >> t;
    while (t--) {
        ll n; cin >> n;
        if (n < 10) {
            cout << n << '\n';
            continue;
        }
        ll b = log10(n);
        ll t = pow(10, b);
        ll a = n / t;
        ll c = n - a * t;
        // a * 10^b + c = n        
        ll ans = b * 9;
        // [t,n]
        ll s = 1;
        for (ll i = 0; i < b; i++) s = s * 10 + 1;
        ll tt = n / s - t / s;
        ans += tt;
        cout << ans << '\n';
    }
    return 0;
}
```

###  #JC0105. A New Technique

---

**分析:** 给定一个矩阵任意序列的行和任意序列的列集合，重组这个矩阵

通过列或行的首位元素即可确定行或列的顺序。

**Code:**

```c++
#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <algorithm>
#include <numeric>
#include <vector>
#include <stdio.h>
#include <cmath>
#include <set>
#include <unordered_map>
using namespace std;
typedef long long ll;
typedef vector<ll> v;
int main() {
    // std::ios::sync_with_stdio(false);
    ll t;
    cin >> t;
    unordered_map<ll, vector<ll>> R;
    vector<ll> RR;
    while (t--) {
        R.clear();
        ll n, m; scanf("%lld%lld", &n, &m);
        for (ll i = 0; i < n; i++) {
            ll r0;
            for (ll j = 0; j < m; j++) {
                ll x; scanf("%lld", &x);
                if (j == 0) {
                    r0 = x; R[r0].resize(m);
                }
                R[r0][j] = x;
            }
        }
        RR.clear();
        bool record = false;
        for (ll i = 0; i < m; i++) {
            for (ll j = 0; j < n; j++) {
                ll x; scanf("%lld", &x);
                if (j == 0 && !record && R.find(x) != R.end())
                    record = true;
                if (record && RR.size() < n)
                    RR.push_back(x);
            }
        }
        for (ll r0 : RR) {
            for (ll r : R[r0]) {
                printf("%lld ", r);
            }
            printf("\n");
        }
    }
    return 0;
}
```

### #JC0107. Collecting Packages

**分析:** 机器人只会↑→移动，故若能收集完毕↑次数，→次数一定；能收集完毕时，以$x$为自变量,$y$一定单调，可以此判断可行性。

字典序最小只需要`R`尽可能在`U`之前即可。

**Code:**

```c++
#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <algorithm>
#include <numeric>
#include <vector>
#include <stdio.h>
#include <cmath>
#include <set>
#include <unordered_map>
using namespace std;
typedef long long ll;
typedef vector<ll> v;
struct P {
    ll x = 0 , y = 0;
};
int main() {
    // std::ios::sync_with_stdio(false);
    ll t;
    cin >> t;
    while (t--) {
        ll n; cin >> n;
        ll xm = 0, ym = 0;
        vector<P> A(n);
        for (auto& p : A) cin >> p.x >> p.y;
        sort(A.begin(), A.end(), [](auto const& lhs, auto const& rhs) {
            if (lhs.x != rhs.x) return lhs.x < rhs.x;
            return lhs.y < rhs.y;
        });
        bool fail = false;
        ll ymax = 0;
        P p0;
        string s;
        for (auto& p : A) {
            ymax = max(ymax, p.y);
            if (ymax > p.y) { fail = true; break; }
            while (p0.x < p.x) p0.x++, s += 'R';
            while (p0.y < p.y) p0.y++, s += 'U';
        }
        if (fail) cout << "NO\n";
        else {
            cout << "YES\n";
            cout << s << '\n';
        }
    }
    return 0;
}
```

### #JC0108. Yet Another Crosses Problem

---

**分析:**找到已有填充的行，列最多的点，在这个点上填充即可

每个点的填充情况需要在线时可知，避免重复填充；直接开`bool[1e5][1e5]`在msvc不魔改`/STACK`会爆栈*(gcc不会！*，堆上用`vector`或`bitset` 可以解决；

**Code:**

```c++
#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <algorithm>
#include <numeric>
#include <vector>
#include <stdio.h>
#include <cmath>
using namespace std;
typedef long long ll;
typedef vector<ll> v;
int main() {
    // std::ios::sync_with_stdio(false);
    ll q; cin >> q;
    vector<vector<bool>> I;
    while (q--) {
        ll n, m; cin >> n >> m;
        getchar();
        v ROW(n),COL(m);       
        I.resize(n);
        for (ll y = 0; y < n; y++) {
            I[y].clear();
            for (ll x = 0; x < m; x++) {
                char c = getchar();
                if (c == '*') {
                    ROW[y]++;
                    COL[x]++;
                }
                I[y].push_back(c == '*');
            }
            getchar();
        }
        ll fills = 1e8;
        for (ll y = 0; y < n; y++) {            
            for (ll x = 0; x < m; x++) {
                bool also_fill = !I[y][x];
                ll col_now = COL[x] + also_fill;
                fills = min(fills, n - ROW[y] + m - col_now);
            }
        }
        printf("%lld\n", fills);
    }
    return 0;
}
```



### #JC0109. RGB Substring (easy version)

---

**分析：**在一个仅由$R,G,B$组成的字符串，修改$m$个字母，让这个字符串存在长$k$，为`"RGBRGBRG..."`子串的字串。

*鉴于数据量较小，就先暴力处理吧（*

对每个符合条件的子串，给定其第一个字母即可确定整个字串的字母；遍历在所有位置上进行改这个位置为`R,G,B`之一字母的所需次数取最小即可。

**Code:**

```c++
#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <algorithm>
#include <numeric>
#include <vector>
#include <stdio.h>
#include <cmath>
using namespace std;
typedef long long ll;
typedef vector<ll> v;
constexpr const char* rgbrgb = "RGBRGB";
ll solve(const char* in, ll length, ll off) {
    ll cnt = 0;
    for (ll i = 0; i < length; i++) {
        char target = *(rgbrgb + (char)((i + off) % 3));
        if (*(in + i) != target) cnt++;
    }
    return cnt;
}
int main() {
    // std::ios::sync_with_stdio(false);
    ll q; cin >> q;
    while (q--) {
        ll n, k; cin >> n >> k;
        string s; cin >> s;        
        ll ans = 1e8;
        for (ll i = 0; i + k <= s.size(); i++) {
            // [i,i+k]
            // -> RGB
            ans = min(ans, solve(s.c_str() + i, k, 0));
            // -> GBR
            ans = min(ans, solve(s.c_str() + i, k, 1));
            // -> BRG
            ans = min(ans, solve(s.c_str() + i, k, 2));
        }
        cout << ans << '\n';
    }
    return 0;
}
```

### #JC0113. 时间复杂度

---

**分析:** byd再做这种码量的模拟我就是[数据删除]

模拟一个栈，几个case*考虑到*即可

- `i` 重复定义 : Crash
- `x` 为 `n`
  - 但 `y` 也为 `n`: continue 
  - `y` 是常数 : break

- `x`,`y`为常数
  - 但 `x<y` : break
  - continue otherwise

- 其他情况继续push

**Code:** 

```c++
#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <algorithm>
#include <numeric>
#include <vector>
#include <stdio.h>
#include <cmath>
#include <set>
using namespace std;
typedef long long ll;
typedef vector<ll> v;
#define PRED(X) [](const auto lhs, const auto rhs) {return X;}
int main() {
    ll t; cin >> t;
    while (t--) {
        ll L; cin >> L;
        string s; cin >> s;
        ll O = 0;
        if (s[2] == 'n')
            O = atoi(s.c_str() + 4);
        vector<std::pair<string,int>> stack;
        int max_c = 0;
        bool exit = false;
        bool error = false;
        while (L--) {            
            string op1, op2, op3, op4;
            auto contrib = [](auto& s1, auto& s2) {
                if (s1 == "n") {
                    if (s2 == "n") return 0;
                    else return -1;
                }
                else if (s2 == "n") return 1;
                else {
                    ll n3 = atoi(s1.c_str()), n4 = atoi(s2.c_str());                   
                    if (n3 > n4) return -1;
                    return 0;
                }
            };
            cin >> op1;
            if (op1 == "F") {                
                cin >> op2 >> op3 >> op4;
                if (find_if(stack.begin(), stack.end(), [&](auto o) { return o.first == op2; }) != stack.end()) {
                    error = true;
                }
                int c = contrib(op3, op4);
                int prev_c = stack.size() ? stack[stack.size() - 1].second : 0;
                if (c == -1 || prev_c == -1) c = -1;
                else c += prev_c;
                max_c = max(c, max_c);
                stack.push_back({ op2,c });
            }
            if (op1 == "E") {              
                if (stack.size()) stack.pop_back();
                else error = true;
            }
        }
        if (stack.size() != 0) error = true;
        if (error) cout << "ERR\n";
        else {
            if (max_c == O) cout << "Yes\n";
            else cout << "No\n";
        }
    }
    return 0;
}
```
